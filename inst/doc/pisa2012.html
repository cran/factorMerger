Its faster to use hclust method on a big dataset.

pisaFMHClustMath <- mergeFactors(pisa2012[, 1:3],
                       factor(pisa2012$CNT),
                       method = "hclust",
                       successive = TRUE) 

plot(pisaFMHClustMath)

pisaFMHClust <- mergeFactors(pisa2012[, 1:3],
                       factor(pisa2012$CNT),
                       method = "hclust",
                       successive = FALSE) 
                       
plot(pisaFMHClust)
